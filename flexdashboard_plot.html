Chart A

ny_noaa %>% 
  filter(month == "01" & year %in% 2000:2007) %>% 
  group_by(year, id) %>%
  summarize(tmax_mean = mean(tmax, na.rm = T)) %>%
  plot_ly(x = ~factor(year), y = ~tmax_mean, type = "box", 
          color = ~factor(year)) %>% 
  layout(title = "Average tmax in January for each year",
         xaxis = list(title = "Year"),
         yaxis = list(title = "Average tmax"))
